```java
private void judge(String token, LinkedList<Character> stack, StringBuilder sb){
    if(token == '('){
        stack.push('(');
    }
    else if(token == ')'){
        while(stack.peek() != '('){
            sb.append(stack.peek());
            stack.pop();
        }
        stack.pop();
    }
    else if(compare(token, stack.peek())){
        sb.
    }
}
```

#### 接雨水（LC42）

> 多解法预警
>
> 暴力法（核心思想，每个位置所能接到的水是左右两边最大高度的较小值再减去自己的高度。）

```java
public void rain1(int[] arr){
    int ans = 0;
    for(int i = 1; i < height.length - 1; ++i){
        int leftMax = 0, rightMax = 0;
        for(int j = 0; j <= i; ++j){
            leftMax = Math.max(leftMax, height[j]);
        }
        for(int j = i; j < height.length; ++j){
            rightMax = Math.max(rightMax, height[j]);
        }
        ans += Math.min(leftMax, rightMax) - height;
    }
    return ans;
}
```

> 双指针，优化动态规划。

```java
public void rain3(int[] arr){
    if(边界条件){
        return ;
    }
    int ans = 0;
    int leftMax = 0, rightMax = 0;
    int left = 0, right = arr.length - 1;
    while(left <= right){
        if(leftMax <= rightMax){
            leftMax = Math.max(leftMax, arr[left]);
            ans += leftMax - arr[left++];
        }
        else{
            rightMax = Math.max(rightMax, arr[right]);
            ans += rightMax - arr[right--];
        }
    }
    return ans;
}
```

> 单调栈

```java
public void rain4(int[] arr){
    if(边界条件){
        return ;
    }
    //这里是要找到右边的比大当前值大的一行一行累加上去
    for(int i = 0; i < height.length; ++i){
        while(!stack.isEmpty() && height[i] > height[stack.peek()]){
            int calIndex = stack.pop();
            if(stack.isEmpty()){
                break;
            }
            int distance = i - stack.peek() - 1;
            int min = Math.min(height[stack.peek()], height[i]);
            ans += distance * (min - height[calIndex]);
        }
        stack.push(i);
    }
    return ans;
}
```

#### 柱状图中的最大的矩形（LC 84）

> 这个和42题最大区别就是一个是递增栈一个是递减栈。

```java
public void maxArea(int[] arr){
    if(边界情况){
        return 边界条件;
    }
    int[] myarr = new int[arr.length + 2];
    System.copyarray(arr, 0, myarr, 2, arr.length);
    LinkedList<Integer> stack = new LinkedList<>();
    int ans = 0;
    for(int i = 0; i < myarr.length; ++i){
        while(!stack.isEmpty() && myarr[stack.peek()] > myarr[i]){
            int curCalIndex = stack.pop();
            ans  = Math.max(ans, myarr[curCalIndex] * (i - stack.peek() - 1))
        }
    }
}
```
